package co.com.elis.core.document;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.LocalTime;
import javax.validation.constraints.NotNull;
import lombok.Getter;

public class InvoicingRangePeriod {

    @Getter
    @NotNull(message = "ELIS_CORE_VAL_INVOICING_PERIOD_START")
    private final LocalDateTime start;

    @Getter
    @NotNull(message = "ELIS_CORE_VAL_INVOICING_PERIOD_END")
    private final LocalDateTime end;

    /**
     * Creates a new Invoicing period between two local date and time
     *
     * @param start Start of the invoicing period
     * @param end End of the invoicing period
     */
    public InvoicingRangePeriod(LocalDateTime start, LocalDateTime end) {
        this.start = start;
        this.end = end;
    }

    /**
     * Creates a new Invoicing period between two local date and time
     *
     * @param startDate Start date of the invoicing period
     * @param startTime Start time of the invoicing period
     * @param endDate End date of the invoicing period
     * @param endTime End time of the invoicing period
     *
     */
    public InvoicingRangePeriod(LocalDate startDate, LocalTime startTime, LocalDate endDate, LocalTime endTime) {
        this.start = startDate.atTime(startTime);
        this.end = endDate.atTime(endTime);
    }

}
